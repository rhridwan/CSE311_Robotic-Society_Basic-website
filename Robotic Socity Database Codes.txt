MEMBER TABLE
CREATE TABLE MEMBER(
    PHONE_NO VARCHAR(15) PRIMARY KEY NOT NULL,
    EMAIL VARCHAR(50) NOT NULL,
    NAME VARCHAR(50) NOT NULL,
    SEX VARCHAR(20),
    PASS VARCHAR(100) NOT NULL,
    INSTITURE VARCHAR(100) NOT NULL
    )

BLOCK TABLE
CREATE TABLE BLOG(
    BLOG_ID VARCHAR(500) PRIMARY KEY NOT NULL,
    ARTICLE LONGTEXT NOT NULL,
    TITTLE VARCHAR(50) NOT NULL,
    DATE VARCHAR(20) NOT NULL,
    TIME VARCHAR(100) NOT NULL
    )

CREATE TABLE ADMIN(
    PASSWORD VARCHAR(500) NOT NULL,
    CONTACT VARCHAR (15) PRIMARY KEY NOT NULL
    )

CREATE TABLE ACHIVEMENT(
    ACHIVEMENT_ID VARCHAR(500) PRIMARY KEY NOT NULL,
    PICTURE VARCHAR (50000) NOT NULL,
    ARTICLE LONGTEXT NOT NULL,
    FLAG VARCHAR(10)  NOT NULL,
    TITTLE VARCHAR(500) NOT NULL
    )


CREATE TABLE CHAT(
    CHAT_ID VARCHAR(500) PRIMARY KEY NOT NULL,
    TEXT LONGTEXT NOT NULL,
    TIME VARCHAR(100) NOT NULL,
    DATE VARCHAR(100) NOT NULL
    )

CREATE TABLE QNA(
    QNA_ID VARCHAR(500) PRIMARY KEY NOT NULL,
    QUES LONGTEXT NOT NULL,
    ANS LONGTEXT NOT NULL,
    TIME VARCHAR(100),
    DATE VARCHAR(100) NOT NULL
    )
